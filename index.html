function calcular() { const dum = new Date(document.getElementById("dum").value); const hoje = new Date(); if (!dum.getTime()) { alert("Selecione uma data válida"); return; } const diff = hoje - dum; const dias = diff / (1000 * 60 * 60 * 24); const semanas = Math.floor(dias / 7); const parto = new Date(dum); parto.setDate(parto.getDate() + 280); let trimestre = ""; let dica = ""; let sintomas = ""; if (semanas <= 12) { trimestre = "🌸 1º Trimestre"; dica = "Ácido fólico e vitaminas pré-natais são os mais comuns nesta fase."; sintomas = "Enjoo, cansaço, sono excessivo, sensibilidade nos seios e alterações hormonais."; } else if (semanas <= 27) { trimestre = "🤰 2º Trimestre"; dica = "Ferro, ômega 3 e hidratação da pele são bastante utilizados."; sintomas = "Melhora dos enjoos, aumento da energia, barriga crescendo e possível dor lombar leve."; } else { trimestre = "🐣 3º Trimestre"; dica = "Produtos para maternidade como fraldas, pomadas e cálcio são comuns."; sintomas = "Cansaço, inchaço nas pernas, falta de ar leve e contrações leves."; } document.getElementById("semanas").innerText = semanas; document.getElementById("trimestre").innerText = trimestre; document.getElementById("dpp").innerText = "Data provável do parto: " + parto.toLocaleDateString("pt-BR"); document.getElementById("dica").innerText = dica; document.getElementById("sintomas").innerText = sintomas; document.getElementById("resultado").classList.remove("hidden"); }
